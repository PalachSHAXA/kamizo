```ts
import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'
import path from 'node:path'

// https://vitejs.dev/config/
export default defineConfig({
  plugins: [react()],
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
    },
  },
  build: {
    outDir: 'public',
    emptyOutDir: false,  // critical — preserves admin.html, icons/, favicon.svg
    rollupOptions: {
      output: {
        manualChunks: {
          gsap: ['gsap'],
          react: ['react', 'react-dom'],
        },
      },
    },
  },
})
```

Why `emptyOutDir: false`: Vite by default wipes `outDir` before build. Our `public/` holds `admin.html`, `icons/`, `favicon.svg` — assets the Worker serves directly. Wiping them on each build breaks the admin panel.

- [ ] **Step 4.1: Write i18n config**

Create `src/i18n/index.ts`:

```ts
import i18n from 'i18next'
import { initReactI18next } from 'react-i18next'
import LanguageDetector from 'i18next-browser-languagedetector'

import ru from './locales/ru.json'
import uz from './locales/uz.json'
import en from './locales/en.json'

export const SUPPORTED_LANGS = ['ru', 'uz', 'en'] as const
export type SupportedLang = (typeof SUPPORTED_LANGS)[number]

i18n
  .use(LanguageDetector)
  .use(initReactI18next)
  .init({
    resources: {
      ru: { translation: ru },
      uz: { translation: uz },
      en: { translation: en },
    },
    fallbackLng: 'ru',
    supportedLngs: SUPPORTED_LANGS,
    interpolation: { escapeValue: false },
    detection: {
      order: ['localStorage', 'navigator'],
      caches: ['localStorage'],
      lookupLocalStorage: 'kamizo-lang',
    },
  })

export default i18n
```

export interface ClientInfo {

  email: string
  company?: string

const API_BASE = ''  // same-origin

export async function getClients(): Promise<ClientInfo[]> {
  const res = await fetch(`${API_BASE}/api/clients`)
  if (!res.ok) throw new Error(`/api/clients ${res.status}`)
  const data = (await res.json()) as ClientInfo[]
  // Hide the 5 hardcoded marketing defaults (Мегаполис/Солнечный/etc.).
  // The user has added real clients (e.g. My Helper) via admin.html — keep those.

export async function verifyDemoCode(code: string): Promise<
  { success: true; demoUrl: string; ukName?: string } | { success: false; error: string }
> {
  const res = await fetch(`${API_BASE}/api/verify-demo`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ code }),
  })
  return (await res.json()) as
    | { success: true; demoUrl: string; ukName?: string }
    | { success: false; error: string }
}

export async function submitDemoRequest(input: DemoRequestInput): Promise<{ success: boolean }> {
  const res = await fetch(`${API_BASE}/api/demo-requests`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ ...input, type: input.type ?? 'demo' }),
  })
  if (!res.ok) throw new Error(`/api/demo-requests ${res.status}`)
  return (await res.json()) as { success: boolean }

```tsx
import { useState, useRef, useEffect } from 'react'
import { X, Send, Check } from 'lucide-react'
import { useTranslation } from 'react-i18next'
import { cn } from '@/lib/utils'
import { submitDemoRequest } from '@/lib/api'

interface ToastProps {
  isOpen: boolean
  onClose: () => void
}

export default function Toast({ isOpen, onClose }: ToastProps) {
  const { t } = useTranslation()
  const [formData, setFormData] = useState({ name: '', phone: '', email: '' })
  const [errors, setErrors] = useState<Record<string, string>>({})
  const [state, setState] = useState<'idle' | 'submitting' | 'success' | 'error'>('idle')
  const toastRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    const onDocClick = (e: MouseEvent) => {
      if (toastRef.current && !toastRef.current.contains(e.target as Node)) onClose()
    }
    if (isOpen) document.addEventListener('mousedown', onDocClick)
    return () => document.removeEventListener('mousedown', onDocClick)
  }, [isOpen, onClose])

  const validate = () => {
    const e: Record<string, string> = {}
    if (!formData.name.trim()) e.name = t('demoForm.errors.name')
    if (!formData.phone.trim()) e.phone = t('demoForm.errors.phone')
    if (!formData.email.trim()) e.email = t('demoForm.errors.emailRequired')
    else if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(formData.email)) e.email = t('demoForm.errors.emailInvalid')
    setErrors(e)
    return Object.keys(e).length === 0
  }

  const onSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    if (!validate()) return
    setState('submitting')
    try {
      await submitDemoRequest({ ...formData, type: 'demo' })
      setState('success')
      setTimeout(() => {
        setState('idle')
        setFormData({ name: '', phone: '', email: '' })
        onClose()
      }, 2500)
    } catch {
      setState('error')
    }
  }

  return (
    <div
      className={cn(
        'fixed bottom-6 right-6 z-[100] w-[380px] max-w-[calc(100vw-48px)] transition-all duration-500',
        isOpen ? 'opacity-100 translate-y-0 pointer-events-auto' : 'opacity-0 translate-y-8 pointer-events-none',
      )}
    >
      <div ref={toastRef} className="bg-white rounded-xl shadow-2xl border border-[#1E1B18]/10 overflow-hidden">
        <div className="flex items-center justify-between p-5 pb-3">
          <div>
            <h3 className="text-lg font-semibold text-[#1E1B18]">{t('demoForm.title')}</h3>
            <p className="text-sm text-[#1E1B18]/50 mt-1">{t('demoForm.subtitle')}</p>
          </div>
          <button onClick={onClose} className="p-1.5 hover:bg-[#1E1B18]/5 rounded-lg transition-colors">
            <X className="w-5 h-5 text-[#1E1B18]/60" />
          </button>
        </div>

        {state !== 'success' ? (
          <form onSubmit={onSubmit} className="p-5 pt-3 space-y-4">
            {(['name', 'phone', 'email'] as const).map((field) => (
              <div key={field}>
                <input
                  type={field === 'email' ? 'email' : field === 'phone' ? 'tel' : 'text'}
                  placeholder={t(`demoForm.placeholder.${field}`)}
                  value={formData[field]}
                  onChange={(e) => setFormData({ ...formData, [field]: e.target.value })}
                  className={cn(
                    'w-full px-4 py-3 bg-[#F8F6F3] rounded-lg text-sm border transition-colors focus:outline-none focus:ring-2 focus:ring-[#E07000]/30',
                    errors[field] ? 'border-red-400' : 'border-transparent',
                  )}
                />
                {errors[field] && <p className="text-xs text-red-500 mt-1">{errors[field]}</p>}
              </div>
            ))}
            <button type="submit" disabled={state === 'submitting'} className="w-full flex items-center justify-center gap-2 px-5 py-3.5 bg-[#E07000] text-white text-sm font-semibold rounded-lg hover:bg-[#c46200] disabled:opacity-60 transition-colors duration-300">
              <Send className="w-4 h-4" />
              {state === 'submitting' ? t('common.loading') : t('common.send')}
            </button>
            {state === 'error' && <p className="text-xs text-red-600 text-center">{t('demoForm.errorSubmit')}</p>}
          </form>
        ) : (
          <div className="p-8 text-center">
            <div className="w-12 h-12 bg-green-100 rounded-full flex items-center justify-center mx-auto mb-4">
              <Check className="w-6 h-6 text-green-600" />
            </div>
            <p className="text-lg font-medium text-[#1E1B18]">{t('demoForm.successTitle')}</p>
            <p className="text-sm text-[#1E1B18]/50 mt-1">{t('demoForm.successDesc')}</p>
          </div>
        )}
      </div>
    </div>
  )
}
```

import { useState } from 'react'
import { X, Key, AlertCircle } from 'lucide-react'
import { useTranslation } from 'react-i18next'
import { cn } from '@/lib/utils'
import { verifyDemoCode } from '@/lib/api'

interface Props {
  isOpen: boolean
  onClose: () => void
  onSwitchToRequest: () => void   // opens the Toast demo form
}

export default function TryDemoModal({ isOpen, onClose, onSwitchToRequest }: Props) {
  const { t } = useTranslation()
  const [code, setCode] = useState('')
  const [state, setState] = useState<'idle' | 'submitting' | 'error'>('idle')
